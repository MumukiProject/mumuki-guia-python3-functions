def half(a_number):
  return a_number / 2

def sum_lengths(a_string, another_string):
  return len(a_string) + len(another_string)
 
def is_question(sentence):
  return str.endswith(sentence, "?")
 
def is_longer_than(a_string, another_string):
  return len(a_string) > len(another_string)
